      Table models:models, each: (model) ->
         [Checkbox model:model, key:'male', 3, new Date()]

Argument strategy

    div ->
      Table ->
        for model in models
          argument ->
            [(Checkbox model:model, key:'male'), 3, -> (div 'asdf')]

      Table ->
        for model in models
          Row ->
            Column ->
              Checkbox model:model, key:'male'
            Column ->
              div 'asdf'

    div ->
      List ->
        for model in models
          argument ->
            Checkbox model:model
